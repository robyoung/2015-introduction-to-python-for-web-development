Decorators

def alert(prefix):
    def decorator(func):
        def inner(*args, **kwargs):
            return "{}! {}".format(prefix,
                                   func(*args, **kwargs))
        return inner
    return decorator

@alert("BOO")
def hello(name):
    return "Hello {}".format(name)

def firstn(n):
    num = 0
    while num < n:
        yield num
        num += 1

Mocking

def something(n):
    return foo.bar(n)

@mock.patch('foo.bar')
def test_something(mock_bar):
    mock_bar.return_value = 12

    assert something(12) == 12

    mock_bar.assert_called_with(12)
